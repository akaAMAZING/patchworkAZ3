    def _generate_league_selfplay(
        self, iteration: int, network_path: Optional[str], output_dir: Path
    ) -> tuple:
        """Generate self-play data using league schedule (multiple opponents).

        Instead of all games using best_model, distributes games across:
          - best_model (stability signal)
          - PFSP-selected opponents (trains against weaknesses)
          - uniform random pool members (broad exploration)

        This ensures the model sees positions from strategies it may be losing to,
        providing the gradient signal needed to beat those strategies.
        """
        num_games = self.selfplay_generator._get_num_games(iteration)
        schedule = self.league.create_schedule(num_games)

        # Build model_schedule: {checkpoint_path: num_games}
        # Multiple opponent IDs may map to the same checkpoint, so we accumulate.
        model_schedule: Dict[str, int] = {}

        # Games vs best model
        if schedule.vs_best > 0:
            best_path = self.league.get_checkpoint_path(self.league.best_id)
            if best_path and Path(best_path).exists():
                model_schedule[best_path] = model_schedule.get(best_path, 0) + schedule.vs_best
            elif network_path:
                model_schedule[network_path] = model_schedule.get(network_path, 0) + schedule.vs_best

        # Games vs PFSP-selected opponents
        for opp_id, games in schedule.vs_pfsp.items():
            opp_path = self.league.get_checkpoint_path(opp_id)
            if opp_path and Path(opp_path).exists():
                model_schedule[opp_path] = model_schedule.get(opp_path, 0) + games

        # Games vs uniform random pool members
        for opp_id, games in schedule.vs_uniform.items():
            opp_path = self.league.get_checkpoint_path(opp_id)
            if opp_path and Path(opp_path).exists():
                model_schedule[opp_path] = model_schedule.get(opp_path, 0) + games

        # Fallback: if all checkpoint paths are missing/invalid, use standard self-play
        if not model_schedule:
            logger.warning("[LEAGUE] No valid pool checkpoints found, falling back to standard self-play")
            if network_path:
                model_schedule[network_path] = num_games
            else:
                return self.selfplay_generator.generate(iteration, network_path, output_dir=output_dir)

        # Log the schedule
        logger.info(
            "[LEAGUE] Self-play schedule: %d games across %d model(s)",
            sum(model_schedule.values()), len(model_schedule),
        )
        for path, n in model_schedule.items():
            logger.info("  %s: %d games", Path(path).stem, n)

        return self.selfplay_generator.generate_with_schedule(iteration, model_schedule, output_dir)

@_njit
def _popcount32(x):
    """Bit population count (Brian Kernighan) — Numba replacement for int.bit_count()."""
    count = 0
    x = x & 0xFFFFFFFF
    while x:
        x = x & (x - 1)
        count += 1
    return count
